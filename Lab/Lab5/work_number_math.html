การทำงานกับ Number และ Math

let a = 10; let b = 3; console.log('a + b =', a + b); // บวก console.log('a % b =', a % b); // เศษจากการหาร console.log('a ** b =', a ** b); // ยกกำลัง // Math Object console.log('ปัดเศษ 3.7:', Math.round(3.7)); console.log('ปัดลง 3.7:', Math.floor(3.7)); console.log('ปัดขึ้น 3.2:', Math.ceil(3.2)); console.log('สุ่ม 0-1:', Math.random()); // สุ่มเลข 1-6 (ลูกเต๋า) let dice = Math.floor(Math.random() * 6) + 1; console.log('ทอยลูกเต๋าได้:', dice);
